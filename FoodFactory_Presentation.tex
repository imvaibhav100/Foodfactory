% SLIDE 6: ARCHITECTURE & DESIGN
% =====================================================
\begin{frame}{Architecture \& Design Patterns}
\begin{block}{MVC Architecture Pattern}
\begin{itemize}
\item \textbf{Model:} Entity classes with JPA annotations
\item \textbf{View:} Thymeleaf templates for dynamic rendering
\item \textbf{Controller:} RESTful controllers handling HTTP requests
\end{itemize}
\end{block}

\begin{block}{Layered Architecture}
\begin{itemize}
\item \textbf{Controller Layer:} Request handling and routing
\item \textbf{Service Layer:} Business logic implementation
\item \textbf{Repository Layer:} Data access with Spring Data JPA
\item \textbf{Entity Layer:} Domain models and database mapping
\end{itemize}
\end{block}

\begin{block}{Key Design Principles}
Separation of Concerns • Dependency Injection • Single Responsibility Principle • RESTful API design
\end{block}
\end{frame}

\begin{frame}{Challenges Faced \& Solutions}
\begin{block}{Security Implementation}
\textbf{Challenge:} Implementing secure authentication and authorization \\
\textcolor{accentcolor}{\textbf{Solution:}} Integrated Spring Security with custom user details service and password encoding

\begin{block}{UI/UX Consistency}
\textbf{Challenge:} Maintaining consistent design across multiple pages \\
\textcolor{accentcolor}{\textbf{Solution:}} Created modular CSS files and reusable Thymeleaf templates
\end{block}

\begin{block}{Database Integration}
\textbf{Challenge:} Managing complex entity relationships \\
\textcolor{accentcolor}{\textbf{Solution:}} Utilized JPA annotations and Spring Data repositories for efficient data handling
\end{block}
